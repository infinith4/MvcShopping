HTML5　Chormeで動作するindexedDB　サンプル１　【入門編　第２回】

Posted 7月 24 2012 by Teachers  HTML5, 技術情報 with 1 コメント

HTML5　Chormeで動作するindexedDB　サンプル１　【入門編　第２回】

indexedDBを調査

HTML5 OFFLINE & STORAGEということで、indexedDBを調査しました。
まず、動くサンプルが殆ど存在しないという難点が。

このIndexedDBはブラウザにNoSQLのDBを持てる仕組みで、FirefoxとChromeで実装されており、IEも10から実装予定のようです。
ただ、Firefoxでは動作が確認できず。

ちなみにWeb SQL DatabaseはSQLiteベースの別の仕組みもあります。
ChromeやSafari、Operaなどで実装されていたのですが、2010年11月に仕様策定が断念されているようです。

今回は、Chrome(20.0.1132.57)にてサンプルを作りました。
Firefox(14.0.1)で確認したところ、以下のエラーがでて全く動きません。
UnknownError: The operation failed for reasons unrelated to the database itself and not covered by any other error code.

indexedDBの概要ですが、まずindexedDBをオープンし、ストアを構築します。
ストアにアクセスする為には、ＤＢよりトランザクションを取得しストアにアクセスします。
それぞれのオブジェクトにアクセスした場合は、そのオブジェクトの処理後に成功/失敗などのコールバック関数を呼び出します。
コードを順をおって見ていきましょう。

indexDB

１．indexedDBをオープンします。

1/* Chromeのみなので、こんな書き方をしなくても良いのですが。 */
2if (window.indexedDB || window.webkitIndexedDB || window.mozIndexedDB) {
3    var indexedDB       = window.indexedDB || window.webkitIndexedDB || window.mozIndexedDB;
4    var IDBTransaction  = window.IDBTransaction || window.webkitIDBTransaction || window.mozIDBTransaction;
5    var IDBKeyRange     = window.IDBKeyRange || window.webkitIDBKeyRange || window.mozIDBKeyRange;
6    var IDBCursor       = window.IDBCursor || window.webkitIDBCursor;
7 
8    /* IndexedDBをオープンします。*/
9    var IDBreq  = indexedDB.open(idbProp.name);
10} else {
11    alert('このブラウザではindexedDB は利用できません');
12}


２．オブジェクト・ストアを作成します。

1/* IndexedDBオープン成功時に呼び出されるコールバック関数 */
2IDBreq.onsuccess    = function (event) {
3    idb = this.result;
4    /* IndexedDBのバージョン・チェック */
5    if (idbProp.ver != idb.version) {
6        /* IndexedDBにバージョンをセット */
7        var req = idb.setVersion(idbProp.ver);
8        /* バージョン・セット成功時に呼び出されるコールバック関数 */
9        req.onsuccess   = function (event) {
10            /* オブジェクト・ストアの作成 */
11            var store   = idb.createObjectStore(
12                idbProp.storeName,
13                { keyPath: 'sKey', autoIncrement: false }
14            );
15            /* 検索キーを作成します */
16            var index   = store.createIndex('indexKey', 'sKey');
17        };
18    }
19};


３．データを登録します。

1/* ボタンを押下しデータを登録します。 */
2$('#set').click(function(){
3    /* トランザクション経由でオブジェクトの保存 */
4    var trans   = idb.transaction(idbProp.storeName, IDBTransaction.READ_WRITE);
5    var store   = trans.objectStore(idbProp.storeName);
6 
7    /* データの内容 */
8    var data    = {
9        sKey : $('#skey').val(),
10        title : $('#title').val(),
11        content : $('#content').val(),
12        postDate : getDate()
13    };
14    /* データをセットします */
15    var res = store.put(data);
16});


４．データを検索します。

1/* ボタンを押下しデータを検索します。 */
2$('#show').click(function(){
3    var key     = $('#getkey').val();
4    /* トランザクション経由でオブジェクトの取得 */
5    var trans   = idb.transaction(idbProp.storeName, IDBTransaction.READ_WRITE);
6    var store   = trans.objectStore(idbProp.storeName);
7    var req     = store.get(key);
8 
9    req.onsuccess = function (event) {
10        if (this.result === undefined) {
11            alert('指定キーで保存されているデータはありません');
12        } else {
13            listout(this.result.sKey, this.result.title, this.result.content, this.result.postDate, false);
14        }
15    };
16});


５．レンジを作成し、データを検索します。

1/* ボタンを押下しデータを検索します。 */
2$('#showIndex').click(function(){
3    /* レンジの作成 */
4    var range   = IDBKeyRange.bound('0', '99999999');
5    var trans   = idb.transaction(idbProp.storeName, IDBTransaction.READ_WRITE);
6    var store   = trans.objectStore(idbProp.storeName);
7 
8    /* インデックスからデータを検索し取得 */
9    var req     = store.openCursor(range, IDBCursor.NEXT);
10    $('#list').html('');
11 
12    req.onsuccess   = function (event) {
13        if (this.result === undefined) {
14            /* データが無い場合の処理 */
15            alert('検索結果がありません。');
16        } else {
17            if(this.result != null){
18                listout(this.result.value.sKey, this.result.value.title, this.result.value.content, this.result.value.postDate, true);
19                /* 次の検索結果でonsuccessコールバック関数を呼び出す */
20                this.result.continue();
21            }
22        }
23    };
24});
